function AvatarShop({student,onUpdate,onBack}){

  const avatar=student.avatar||{gender:"male",kit:"basic",hair:"short",extra:false};

  const extraItem=isMale?ARMBAND:TIARA_ITEM;
  function buy(id,cost,type){
    if(coins<cost){setMsg("Ikke nok coins!");setTimeout(()=>setMsg(""),2000);return;}
    if(purchased.includes(id)){setMsg("Allerede kjøpt!");setTimeout(()=>setMsg(""),1500);return;}

      </div>
      {msg&&<div style={{background:"#e8f5e9",borderRadius:10,padding:"10px",marginBottom:10,fontSize:14,fontWeight:700,color:"#2e7d32"}}>{msg}</div>}
      <div style={{display:"flex",gap:6,marginBottom:14,flexWrap:"wrap"}}>

function PackStore({student,onUpdate,onBack}){
  const [phase,setPhase]=useState("store");
  const [selPack,setSelPack]=useState(null);
  const [revealed,setRevealed]=useState([]);
  const [revIdx,setRevIdx]=useState(0);
  const [msg,setMsg]=useState("");
  const coins=student.coins||0,owned=student.playerCards||[];
  function buyPack(pack){
    if(coins<pack.price){setMsg("Ikke nok coins! 🪙");setTimeout(()=>setMsg(""),2000);return;}
    const players=openPack(pack);
    onUpdate({coins:coins-pack.price});
    setSelPack(pack);setRevealed(players);setRevIdx(0);setPhase("opening");
  }
  function revealNext(){
    const p=revealed[revIdx];
    const ex=student.playerCards||[];
    if(!ex.includes(p.id))onUpdate({playerCards:[...ex,p.id]});
    if(revIdx+1>=revealed.length)setPhase("reveal");
    else setRevIdx(i=>i+1);
  }
  if(phase==="opening"){
    const cur=revealed[revIdx];
    return(
      <div style={s.page}>
        <div style={{textAlign:"center",padding:"20px 0"}}>
          <div style={{fontSize:13,color:"var(--t3)",marginBottom:4}}>{selPack.label} · Kort {revIdx+1} av {revealed.length}</div>

          <button style={s.bigBtn} onClick={revealNext}>{revIdx+1<revealed.length?"Neste kort ▶":"Se alle ✓"}</button>
        </div>
      </div>
    );

  if(phase==="reveal")return(
    <div style={s.page}>
      <div style={{textAlign:"center",padding:"16px 0 8px"}}><div style={{fontSize:40}}>🎉</div><div style={{fontSize:20,fontWeight:800,margin:"8px 0"}}>Pakke åpnet!</div></div>

    </div>
  );
  return(
    <div style={s.page}>

// ── FLAG QUIZ ──
function FlagQuiz({onBack,addCoins,onDone}){
  const [deck]=useState(()=>getDailyFlags());
  const [idx,setIdx]=useState(0);
  const [chosen,setChosen]=useState(null);
  const [score,setScore]=useState(0);
  const [done,setDone]=useState(false);
  const [choices,setChoices]=useState([]);
  function makeChoices(i){
    const correct=deck[i];
    setChoices(shuffle([correct,...shuffle(FLAGS.filter(f=>f.code!==correct.code)).slice(0,3)]));
    setChosen(null);
  }
  useEffect(()=>{makeChoices(0);},[]);
  function pick(flag){
    if(chosen)return;setChosen(flag.code);
    const ok=flag.code===deck[idx].code;
    if(ok){setScore(sc=>sc+1);if(addCoins)addCoins(1);}
    setTimeout(()=>{
      if(idx+1>=deck.length){if(onDone)onDone();setDone(true);}
      else{setIdx(i=>{makeChoices(i+1);return i+1;});}
    },1000);
  }
  if(done)return(
    <div style={s.page}><div style={s.rb}>
      <div style={{fontSize:64}}>{score>=18?"🏆":score>=12?"🌟":"💪"}</div>
      <div style={s.rt}>Flaggquiz ferdig!</div>
      <div style={s.rs}>{score} av {deck.length} riktige</div>
      <div style={{fontSize:16,color:"#f9a825",fontWeight:700}}>+{score} 🪙</div>
      <div style={{fontSize:13,color:"var(--t3)"}}>Nye flagg i morgen!</div>
      <button style={{...s.bigBtn,...s.btnG}} onClick={onBack}>Hjem</button>
    </div></div>
  );
  const cur=deck[idx];
  return(
    <div style={s.page}>
      <div style={s.tb}><button style={s.back} onClick={onBack}>← Hjem</button><span style={s.tt}>🌍 Flaggquiz</span><span style={{fontSize:14,fontWeight:700}}>⭐{score}</span></div>
      <div style={s.prog}>{idx+1}/{deck.length}</div>
      <div style={{textAlign:"center",marginBottom:20}}>
        <div style={{fontSize:13,color:"var(--t3)",marginBottom:12}}>Hvilket land er dette?</div>
        <img src={`https://flagcdn.com/w160/${cur.code.toLowerCase()}.png`} alt={cur.code} style={{width:160,height:100,objectFit:"cover",borderRadius:10,border:"1px solid #ddd",boxShadow:"0 4px 12px rgba(0,0,0,0.15)"}}/>
      </div>
      <div style={s.cg}>
        {choices.map((f,i)=>{
          let bg="#fff",border="1px solid #ddd";
          if(chosen){if(f.code===cur.code){bg="#d4f7d4";border="2px solid #43a047";}else if(f.code===chosen){bg="#fdd";border="2px solid #e53935";}}
          return<button key={i} style={{...s.choice,background:bg,border,fontSize:14,padding:"16px 8px",fontWeight:600}} onClick={()=>pick(f)}>{f.name}</button>;

function HangmanGame({gloser,onBack,onCorrect,onDone}){
  const [deck]=useState(()=>shuffle(gloser));
  const [idx,setIdx]=useState(0);
  const [guessed,setGuessed]=useState(new Set());
  const [score,setScore]=useState(0);
  const [done,setDone]=useState(false);
  const word=deck[idx]?.en||"",hint=deck[idx]?.no||"";
  const letters=word.split("");
  const wrong=[...guessed].filter(l=>!letters.includes(l));
  const wc=wrong.length,won=letters.every(l=>guessed.has(l)),lost=wc>=MAX_WRONG,fin=won||lost;
  function guess(l){
    if(fin||guessed.has(l))return;
    const nx=new Set(guessed);nx.add(l);setGuessed(nx);
    if(letters.every(c=>nx.has(c))){setScore(sc=>sc+1);if(onCorrect)onCorrect();}
  }
  function next(){
    if(idx+1>=deck.length){if(onDone)onDone();setDone(true);return;}
    setIdx(i=>i+1);setGuessed(new Set());
  }
  const alpha="abcdefghijklmnopqrstuvwxyz".split("");
  if(done)return(
    <div style={s.page}><div style={s.rb}>
      <div style={{fontSize:64}}>{score===deck.length?"🏆":score>=deck.length*0.6?"🌟":"💪"}</div>

      <div style={{fontSize:16,color:"#f9a825",fontWeight:700}}>+{score} 🪙</div>
      <button style={s.bigBtn} onClick={()=>{setIdx(0);setGuessed(new Set());setScore(0);setDone(false);}}>Prøv igjen</button>
      <button style={{...s.bigBtn,...s.btnG}} onClick={onBack}>Hjem</button>
    </div></div>
  );
  return(
    <div style={s.page}>
      <div style={s.tb}><button style={s.back} onClick={onBack}>← Hjem</button><span style={s.tt}>Hangman</span><span style={{fontSize:14,fontWeight:700}}>⭐{score}</span></div>

        <svg viewBox="0 0 120 130" width="120" height="130">

          <line x1="30" y1="125" x2="30" y2="10" stroke="#555" strokeWidth="3" strokeLinecap="round"/>
          <line x1="30" y1="10" x2="75" y2="10" stroke="#555" strokeWidth="3" strokeLinecap="round"/>
          <line x1="75" y1="10" x2="75" y2="25" stroke="#555" strokeWidth="3" strokeLinecap="round"/>
          {wc>=1&&<circle cx="75" cy="33" r="8" stroke="#e53935" strokeWidth="2.5" fill="none"/>}
          {wc>=2&&<><line x1="75" y1="41" x2="75" y2="70" stroke="#e53935" strokeWidth="2.5" strokeLinecap="round"/><line x1="75" y1="50" x2="58" y2="62" stroke="#e53935" strokeWidth="2.5" strokeLinecap="round"/><line x1="75" y1="50" x2="92" y2="62" stroke="#e53935" strokeWidth="2.5" strokeLinecap="round"/></>}
          {wc>=3&&<><line x1="75" y1="70" x2="62" y2="88" stroke="#e53935" strokeWidth="2.5" strokeLinecap="round"/><line x1="75" y1="70" x2="88" y2="88" stroke="#e53935" strokeWidth="2.5" strokeLinecap="round"/></>}
        </svg>
      </div>

// ── MATHS ──
function MathGame({onBack,addCoins,mode,onDone}){
  const [score,setScore]=useState(0);
  const [q,setQ]=useState(null);
  const [chosen,setChosen]=useState(null);
  const [done,setDone]=useState(false);
  const [round,setRound]=useState(0);
  const TOTAL=10;
  function newQ(){
    let a,b,answer,op;
    if(mode==="multiply"){a=Math.floor(Math.random()*10)+1;b=Math.floor(Math.random()*10)+1;answer=a*b;op="×";}
    else if(mode==="add"){a=Math.floor(Math.random()*20)+1;b=Math.floor(Math.random()*20)+1;answer=a+b;op="+";}
    else{a=Math.floor(Math.random()*20)+10;b=Math.floor(Math.random()*a)+1;answer=a-b;op="−";}
    const wrong=new Set();
    while(wrong.size<3){let w=answer+(Math.floor(Math.random()*8)-4);if(w!==answer&&w>0)wrong.add(w);}
    setQ({a,b,answer,op,choices:shuffle([answer,...wrong])});setChosen(null);
  }
  useEffect(()=>{newQ();},[]);
  function pick(val){
    if(chosen!==null)return;setChosen(val);
    if(val===q.answer){setScore(sc=>sc+1);if(addCoins)addCoins(1);}
    setTimeout(()=>{
      if(round+1>=TOTAL){if(onDone)onDone();setDone(true);}
      else{setRound(r=>r+1);newQ();}
    },900);
  }
  const colors={multiply:"#e0f7fa",add:"#e8f5e9",subtract:"#fff3e0"};
  const titles={multiply:"✖️ Ganging",add:"➕ Addisjon",subtract:"➖ Subtraksjon"};
  if(done)return(
    <div style={s.page}><div style={s.rb}>
      <div style={{fontSize:64}}>{score>=9?"🏆":score>=6?"🌟":"💪"}</div>
      <div style={s.rt}>Ferdig!</div><div style={s.rs}>{score} av {TOTAL}</div>
      <div style={{fontSize:16,color:"#f9a825",fontWeight:700}}>+{score} 🪙</div>
      <button style={s.bigBtn} onClick={()=>{setScore(0);setRound(0);setDone(false);newQ();}}>Prøv igjen</button>
      <button style={{...s.bigBtn,...s.btnG}} onClick={onBack}>Hjem</button>
    </div></div>
  );
  if(!q)return null;
  return(
    <div style={s.page}>
      <div style={s.tb}><button style={s.back} onClick={onBack}>← Hjem</button><span style={s.tt}>{titles[mode]}</span><span style={{fontSize:14,fontWeight:700}}>⭐{score}</span></div>
      <div style={s.prog}>{round+1}/{TOTAL}</div>
      <div style={{background:colors[mode],borderRadius:20,padding:"36px 24px",textAlign:"center",marginBottom:20}}>
        <div style={{fontSize:13,color:"var(--t3)",marginBottom:6}}>Hva er?</div>
        <div style={{fontSize:50,fontWeight:800}}>{q.a} {q.op} {q.b} = ?</div>
      </div>
      <div style={s.cg}>
        {q.choices.map((c,i)=>{let bg="#fff";if(chosen!==null){if(c===q.answer)bg="#d4f7d4";else if(c===chosen)bg="#fdd";}return<button key={i} style={{...s.choice,background:bg,fontSize:26,fontWeight:700,padding:"20px 8px"}} onClick={()=>pick(c)}>{c}</button>;})}
      </div>
    </div>
  );
}

// ── PENALTY ──
function PenaltyGame({onDone}){
  const [phase,setPhase]=useState("aim");
  const [target,setTarget]=useState({x:50,y:50});
  const [scored,setScored]=useState(null);
  const [shots,setShots]=useState([]);
  const [current,setCurrent]=useState(0);
  const TOTAL=5;
  function shoot(){
    if(phase!=="aim")return;
    const keeperX=Math.random()>0.5?20:80;
    const hit=Math.abs(target.x-keeperX)>25||target.y<30;
    setPhase("flying");
    setTimeout(()=>{const ns=[...shots,hit];setShots(ns);setScored(hit);setPhase("result");setTimeout(()=>{if(current+1>=TOTAL){onDone(ns.filter(Boolean).length);}else{setCurrent(c=>c+1);setScored(null);setPhase("aim");setTarget({x:50,y:50});}},1200);},800);
  }
  return(
    <div style={{textAlign:"center",padding:"16px 0"}}>
      <div style={{fontSize:13,fontWeight:700,marginBottom:10}}>Skytt {current+1} av {TOTAL} · Mål: {shots.filter(Boolean).length}</div>
      <div style={{position:"relative",width:"100%",maxWidth:320,margin:"0 auto",height:180,background:"linear-gradient(180deg,#90ee90,#228b22)",borderRadius:12,overflow:"hidden",cursor:phase==="aim"?"crosshair":"default",border:"2px solid #555"}}
        onClick={e=>{if(phase!=="aim")return;const r=e.currentTarget.getBoundingClientRect();setTarget({x:((e.clientX-r.left)/r.width)*100,y:((e.clientY-r.top)/r.height)*100});}}>
        <div style={{position:"absolute",top:10,left:"15%",width:"70%",height:"55%",border:"4px solid white",borderBottom:"none",pointerEvents:"none"}}/>
        {phase==="aim"&&<div style={{position:"absolute",left:`${target.x}%`,top:`${target.y}%`,transform:"translate(-50%,-50%)",fontSize:22,pointerEvents:"none"}}>⚽</div>}
        {phase==="result"&&scored!==null&&<div style={{position:"absolute",inset:0,display:"flex",alignItems:"center",justifyContent:"center",background:scored?"rgba(0,180,0,0.5)":"rgba(200,0,0,0.5)",fontSize:32}}>{scored?"⚽ MÅL!":"❌ Reddet!"}</div>}
        {phase==="aim"&&<div style={{position:"absolute",bottom:6,width:"100%",textAlign:"center",fontSize:11,color:"rgba(255,255,255,0.8)"}}>Trykk der du vil skyte</div>}
      </div>
      {phase==="aim"&&<button style={{...s.bigBtn,marginTop:14,background:"#e53935"}} onClick={shoot}>🦵 Skyt!</button>}
      <div style={{display:"flex",justifyContent:"center",gap:8,marginTop:10}}>
        {Array(TOTAL).fill(0).map((_,i)=><span key={i} style={{fontSize:18}}>{i<shots.length?(shots[i]?"⚽":"❌"):"⬜"}</span>)}
      </div>
    </div>
  );
}
